  <script>
    function calculate() {
      const acc = parseFloat(document.getElementById("account").value);
      const riskP = parseFloat(document.getElementById("risk").value);
      const entry = parseFloat(document.getElementById("entry").value);
      const stop = parseFloat(document.getElementById("stop").value);
      const tp = parseFloat(document.getElementById("tp").value);

      const res = document.getElementById("result");

      if ([acc, riskP, entry, stop, tp].some(isNaN)) {
        res.textContent = "ကျေးဇူးပြုပြီး အချက်အလက်အားလုံးဖြည့်ပါ။";
        return;
      }

      const riskAmount = acc * (riskP / 100);
      const lossPerUnit = entry - stop;
      if (lossPerUnit <= 0) {
        res.textContent = "Stop Loss သည် Entry ထက် နိမ့်ရမည်။";
        return;
      }

      const units = riskAmount / lossPerUnit;
      const rewardPerUnit = tp - entry;
      const rr = rewardPerUnit / lossPerUnit;

      res.innerHTML = `
        💵 အန္တရာယ်ငွေ (Risk): <b>$${riskAmount.toFixed(2)}</b><br>
        📊 ဝယ်မည့်အချိုး (Position Size): <b>${units.toFixed(2)} Units</b><br>
        ⚖️ R:R Ratio: <b>${rr.toFixed(2)} : 1</b><br>
        ✅ Take Profit အကြံပြုချက်: <b>${tp}</b>
      `;
    }
  </script>
